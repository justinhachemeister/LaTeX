	\dfn{$\bs{q-}$Pochhammer Symbol}{\begin{align*}
			&(a)_0\coloneqq (a;q)_0\coloneqq 1\\
			&(a)_n\coloneqq (a;q)_n\coloneqq \prod_{k=0}^{n-1}\lt(1-aq^k\rt), && n\geq 1\\
			&(a)_{\infty}n\coloneqq (a;q)_{\infty}\coloneqq \prod_{k=0}^{\infty}\lt(1-aq^k\rt), && |q|<1
	\end{align*}Here $q$ is called the base and if the identification of the base is clear, we often omit $q$ from the notation.}
